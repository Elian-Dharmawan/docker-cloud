    #!/bin/bash
    sudo yum update -y
    sudo yum install curl -y
    sudo amazon-linux-extras install docker -y
    sudo yum install git -y
    sudo mkdir /var/lib/efs
    sudo mount -t nfs4 -o nfsvers=4.1,rsize=1048576,wsize=1048576,hard,timeo=600,retrans=2,noresvport fs-01d133fdd3b524dc1.efs.us-east-1.amazonaws.com:/ /var/lib/efs
    sudo -- bash -c 'echo "fs-01d133fdd3b524dc1.efs.us-east-1.amazonaws.com:/ /var/lib/efs nfs4 defaults,_netdev 0 0" >> /etc/fstab'
    sudo systemctl enable docker
    sudo service docker start
    sudo usermod -a -G docker ec2-user
    sudo curl -L "https://github.com/docker/compose/releases/download/1.29.2/docker-compose-$(uname -s)-$(uname -m)" -o /usr/local/bin/docker-compose
    sudo chmod +x /usr/local/bin/docker-compose
    sudo mkdir /home/ec2-user/wordpres
    cd /home/ec2-user/wordpres/
    git clone https://github.com/Elian-Dharmawan/docker-cloud.git
    cd ./docker-cloud
    wget https://tugas-final-cloud.s3.amazonaws.com/.env
    docker-compose up -d
